Callback и чейнинг

Как раньше отправляли запросы:

http.get('/api/v1/chats', function (err, chats) {
  if (err) {
	  console.error(err);
	  return;
  }
  console.log('chats: ', chats);
});
 
console.log('Waiting...');

Вроде бы всё хорошо. А теперь рассмотрим классический случай — нужно авторизоваться, получить чаты и только потом данные пользователя:

http.post('/api/v1/signin', user, function (err, resp1) {
  if (err) { return console.error(err); }
  http.get(`/api/v1/chats/${resp1.id}`, function (err, resp2) {
    if (err) { return console.error(err); }
    http.post(`/api/v1/chats/${resp1.id}/messages`, {info: resp2}, function (err, avatar) {
      if (err) { return console.error(err); }
      // callback hell
    });
  });
});

Вложенность запросов могла разрастаться до невероятных глубин, и в итоге получались даже такие пирамиды:

Их называют “callback hell” — адом колбэков. Разобраться, что происходит в таком коде — испытание на мужество и усидчивость. Разработчики потеряли не одну сотню нервных клеток во время дебага таких конструкций.

Избавление от вложенности

Как сберечь нервы и не попасть в ад колбэков? Для начала выделим колбэки в отдельные функции, которые передадим в аргумент через имя функции:

http.get('/api/v1/chats', handleLoadChats);

function handleLoadChats(err, chats) {
  if (err) {
	  console.error(err);
	  return;
  }
  console.log('chats: ', chats);
}

Можно выбрасывать исключения, чтобы не проваливаться дальше:

try {
  // Выбрасываем исключение вручную
  throw new Error('Error!');
} catch (err) {
  alert(err); // Error!
}

Применим подобную технику для примера про получение чатов и информации о пользователе:

const callback = function (err, chats) {
  if (err) {
    throw err;
  }
  console.log('Chats: ', chats);
};
 
try {
  http.get('/user', callback);
} catch (err) {
  console.log(err); // Не выполнится
}

Глаза радуются, душа спокойна, а читать такой код — сплошное удовольствие. Особенно если познавать этот подход в сравнении с «адским» вариантом.
