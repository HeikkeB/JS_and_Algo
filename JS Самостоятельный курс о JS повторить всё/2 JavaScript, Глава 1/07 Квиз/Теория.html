5. Что будет выведено в консоль?

function checkAnimal(data) {
  if (data === { animal: 'Кот' }) {
    console.log('Это кот');
  } else if (data == { animal: 'Кот' }) {
    console.log('Всё ещё кот');
  } else {
    console.log('Я таких животных не знаю');
  }
}

checkAnimal({ animal: 'Кот' });

  • 'Это кот'

  • 'Всё ещё кот'

  • 'Я таких животных не знаю'
